Условие:
В массиве представлены средние скорости движения трафика на улицах города (всего N улиц). Необходимо выяснить, какие улицы самые высоконагруженные, и вывести номера улиц, по которым можно пускать маршруты в объезд.

Тоесть
Чем меньше скорость движения, тем больше загружена улица
Для объезда берем улицы, где скорость выше средней



#include <iostream>
using namespace std;
int main()
{
    int N;
    cout << "Введите кол-во улиц: ";
    cin >> N;

    float speeds[100]; // массив со скоростями, максимум 100 улиц

    for (int i = 0; i < N; i++) // вводим скорости на улицах
    {
        cout << "Cкорость на улице: ";
        cin >> speeds[i];
    }

    float min = speeds[0]; // минимум ищем для самой загруженной улицы
    for (int i = 0; i < N; i++)
    {
        if (speeds[i] < min)
        {
            min = speeds[i];
        }
    }

    float sum = 0; // сумма всех скоростей
    for (int i = 0; i < N; i++)
        sum += speeds[i];

    float avg = sum / N; // средняя скорость

    cout << "\nСамые высоконагруженные улицы: ";
    for (int i = 0; i < N; i++) // выводим самые загруженные улицы
    {
        if (speeds[i] == min)
            cout << "Улица " << i + 1 << " ";
    }

    cout << "\nУлицы для объезда: ";
    for (int i = 0; i < N; i++) // выводим улицы со скоростью выше средней
    {
        if (speeds[i] > avg)
            cout << "Улица " << i + 1 << " ";
    }

    return 0;
}

Условие:
Дана строка. Расставить правильно запятые перед союзами «а» и «но».

Изза проблем с русскими символами в string использую английское предложение
перед отдельными словами a и but ставим запятую если ее нет


#include <iostream>
#include <string>
using namespace std;
int main()
{
    string s = "I went home a he stayed but then left";

    for (int i = 0; i < s.length(); i++) // идем по буквам
    {
        if (s[i] == 'a') // нашли a
        {
            if ((i == 0 || s[i - 1] == ' ') &&
                (i + 1 == s.length() || s[i + 1] == ' '))
            {
                // (стоим на 1-м символе или прошлый символ — пробел)
                // и
                // (стоим на последнем символе слова или следующий символ — пробел)
                // 1-я скобка: слева от буквы начало слова
                // 2-я скобка: справа конец слова

                if (!((i > 0 && s[i - 1] == ',') ||
                      (i > 1 && s[i - 2] == ',' && s[i - 1] == ' ')))
                {
                    // если перед словом нет запятой — вставляем ее
                    // 1-я скобка: перед словом уже стоит запятая
                    // 2-я скобка: перед словом стоит пробел, а перед пробелом запятая
                    // все это отрицаем, значит запятой перед словом нет

                    s.insert(i-1, ","); // ставим запятую перед словом a
                    i++;
                }
            }
        }

        if (s.substr(i, 3) == "but") // нашли but
        {
            if ((i == 0 || s[i - 1] == ' ') &&
                (i + 3 == s.length() || s[i + 3] == ' '))
            {
                // (стоим на первом символе слова but, и слева либо начало строки, либо пробел)
                // и
                // (после слова but либо конец строки, либо пробел)
                // тут +3, потому что but состоит из 3 символов

                if (!((i > 0 && s[i - 1] == ',') ||
                      (i > 1 && s[i - 2] == ',' && s[i - 1] == ' ')))
                {
                    // если перед словом but нет запятой — вставляем ее
                    // 1-я скобка: перед словом стоит запятая
                    // 2-я скобка: случай вида xx, but
                    // все это отрицаем

                    s.insert(i-1, ","); // ставим запятую перед словом but
                    i++;
                }
            }
        }
    }

    cout << s << endl;

    return 0;
}
