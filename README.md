Условие:
В массиве представлены средние скорости движения трафика на улицах города (всего N улиц). Необходимо выяснить, какие улицы самые высоконагруженные, и вывести номера улиц, по которым можно пускать маршруты в объезд.

Тоесть
Чем меньше скорость движения, тем больше загружена улица
Для объезда берем улицы, где скорость выше средней


```cpp
#include <iostream>
using namespace std;
int main()
{
    int N;
    cout << "Введите кол-во улиц: ";
    cin >> N;

    float speeds[N]; // колво улиц - N ; массив со скоростями

    for (int i = 0; i < N; i++) // вводим наши скорости на улицах
    {
        cout << "Cкорость на улице: ";
        cin >> speeds[i];
    }

    float min = speeds[0]; // минимум ищем для самой заторможенной
    for (int i = 0; i < N; i++)
    {
        if (speeds[i] < min)
        {
            min = speeds[i];
        }
    }

    float sum = 0; // для ср скорости
    for (int i = 0; i < N; i++)
        sum += speeds[i];
    float avg = sum / N;

    cout << "\nСамые высоконагруженные улицы: "; // выводим самые загруженные
    for (int i = 0; i < N; i++)
    {
        if (speeds[i] == min)
            cout << "Улица " << i + 1 << " ";
    }

    cout << "\nУлицы для объезда: "; // выводим самые не загруженные
    for (int i = 0; i < N; i++) 
    {
        if (speeds[i] > avg)
            cout << "Улица " << i + 1 << " ";
    }

    return 0;
}
```

```text
Номер 1 переделанный с сортировкой методом пузырька, 
тут после сортировки ищется максимальный разрыв между соседними скоростями (элемент i - (i+1) ) .
Улицы с минимальными скоростями считаются самыми загруженными, остальные рукомендую в объезд
```
```cpp

#include <iostream>

using namespace std;

int main()
{
    int N;
    cout << "Вводим колво улиц: ";
    cin >> N;

    float speeds[N]; //массив скоростей
    int numbers[N]; // номера улиц

    for (int i = 0; i < N; i++) //цикл по вводу
    {
        cout << "Скорость на улице: ";
        cin >> speeds[i]; // закидываем скорости
        numbers[i] = i + 1; //закидываем номера улиц
    }

    // сортировка пузырьком взял из лекции
    for (int a = 1; a < N; a++)
    {
        for (int b = N - 1; b >= a; b--) // идем с конца к началу
        {
            if (speeds[b - 1] > speeds[b]) // левый элт больше правого — меняем местами
            {
                float tempSpeed = speeds[b - 1]; // меняем местами два соседних элта
                speeds[b - 1] = speeds[b];
                speeds[b] = tempSpeed;

                int tempNumber = numbers[b - 1]; // меняем местаии номера улиц чтобы не потеряться
                numbers[b - 1] = numbers[b];
                numbers[b] = tempNumber;
            }
        }
    }

    if (N == 1) // ток одна улица
    {
        cout << "\nСамая высоконагруженная улица: " << numbers[0];
        cout << "\nУлиц для объезда нет";
        return 0;
    }

    float maxDiff = speeds[1] - speeds[0]; // разница между двумя элементами (1 и 2)
    int border = 0; // индекс границы между группами

    for (int i = 0; i < N - 1; i++)
    {
        float diff = speeds[i + 1] - speeds[i]; // разница между соседними элтами
        if (diff > maxDiff) // если разница больше текущего максимального
        {
            maxDiff = diff; // запоминаем новый максимум и позицию
            border = i;
        }
    }

    cout << "\nСамые высоконагруженные улицы: ";
    for (int i = 0; i <= border; i++) //берем эл-ты до границы
    {
        cout << numbers[i] << " ";  // вывод номеров улиц
    }

    cout << "\nУлицы для объезда: ";
    for (int i = border + 1; i < N; i++) //элты после границы
    {
        cout << numbers[i] << " ";
    }

    return 0;
}
```

```text

Условие:
Дана строка. Расставить правильно запятые перед союзами «а» и «но».

Почему английское предложение: изза проблем с русскими символами в string использую английское предложение
перед отдельными словами a и but ставим запятую если ее нет
```
```cpp
#include <iostream>
#include <string>
using namespace std;
int main()
{
    string s = "I went home a he stayed but then left";

    for (int i = 0; i < s.length(); i++) // идем по буквам
    {
        if (s[i] == 'a') // нашли a
        {
            if ((i == 0 || s[i - 1] == ' ') &&
                (i + 1 == s.length() || s[i + 1] == ' '))
            {
                //(стоим на 1м символе или прошлый - пробел) и (стоим на последнем символе или следующий - пробел
                // 1е усл в скобках чтобы за пределы не уйти
                // (стоим на последнем символе слова или следующий символ — пробел)
                // 1 скобка - слева от буквы - начало слова ; 2 - справа конец

                if (!((i > 0 && s[i - 1] == ',') ||
                      (i > 1 && s[i - 2] == ',' && s[i - 1] == ' ')))
                {
                    // если перед словом нет , - вставляем ,
                    // 1я скобка - перед символом стоит запятая
                    // 2я скобка - перед текущим символом стоит пробел и перед ним запятая
                    // и мы отрицаем все это дело

                    s.insert(i-1, ","); // вот так оно выглядит xx, а
                    i++;
                }
            }
        }

        if (s.substr(i, 3) == "but") // нашли but
        {
            if ((i == 0 || s[i - 1] == ' ') &&
                (i + 3 == s.length() || s[i + 3] == ' '))
            {
                // стоим на первом символе but, и слева либо начало строки, либо пробел
                // и (после слова but либо конец строки, либо пробел)
                // тут +3, тк but из 3 символов

                if (!((i > 0 && s[i - 1] == ',') ||
                      (i > 1 && s[i - 2] == ',' && s[i - 1] == ' ')))
                {
                    // если перед словом but нет запятой — вставляем ее
                    // 1-я скобка: перед but сразу стоит запятая
                    // 2-я скобка = если там xx, but
                    // и мы отрицаем все это дело

                    s.insert(i-1, ","); // чтобы выглядело так: xx, but
                    i++;
                }
            }
        }
    }

    cout << s << endl;

    return 0;
}
```
