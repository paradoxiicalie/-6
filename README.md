Номер 1

Условие:
В массиве представлены средние скорости движения трафика на улицах города (всего N улиц). Необходимо выяснить, какие улицы самые высоконагруженные, и вывести номера улиц, по которым можно пускать маршруты в объезд.

Тоесть
Чем меньше скорость движения, тем больше загружена улица
Для объезда берем улицы, где скорость выше средней


```cpp
#include <iostream>
using namespace std;
int main()
{
    int N;
    cout << "Введите кол-во улиц: ";
    cin >> N;

    float speeds[N]; // колво улиц - N ; массив со скоростями

    for (int i = 0; i < N; i++) // вводим наши скорости на улицах
    {
        cout << "Cкорость на улице: ";
        cin >> speeds[i];
    }

    float min = speeds[0]; // минимум ищем для самой заторможенной
    for (int i = 0; i < N; i++)
    {
        if (speeds[i] < min)
        {
            min = speeds[i];
        }
    }

    float sum = 0; // для ср скорости
    for (int i = 0; i < N; i++)
        sum += speeds[i];
    float avg = sum / N;

    cout << "\nСамые высоконагруженные улицы: "; // выводим самые загруженные
    for (int i = 0; i < N; i++)
    {
        if (speeds[i] == min)
            cout << "Улица " << i + 1 << " ";
    }

    cout << "\nУлицы для объезда: "; // выводим самые не загруженные
    for (int i = 0; i < N; i++) 
    {
        if (speeds[i] > avg)
            cout << "Улица " << i + 1 << " ";
    }

    return 0;
}
```

Номер 2

Условие:
Дана строка. Расставить правильно запятые перед союзами «а» и «но».

Почему английское предложение: изза проблем с русскими символами в string использую английское предложение
перед отдельными словами a и but ставим запятую если ее нет

```cpp
#include <iostream>
#include <string>
using namespace std;
int main()
{
    string s = "I went home a he stayed but then left";

    for (int i = 0; i < s.length(); i++) // идем по буквам
    {
        if (s[i] == 'a') // нашли a
        {
            if ((i == 0 || s[i - 1] == ' ') &&
                (i + 1 == s.length() || s[i + 1] == ' '))
            {
                //(стоим на 1м символе или прошлый - пробел) и (стоим на последнем символе или следующий - пробел
                // 1е усл в скобках чтобы за пределы не уйти
                // (стоим на последнем символе слова или следующий символ — пробел)
                // 1 скобка - слева от буквы - начало слова ; 2 - справа конец

                if (!((i > 0 && s[i - 1] == ',') ||
                      (i > 1 && s[i - 2] == ',' && s[i - 1] == ' ')))
                {
                    // если перед словом нет , - вставляем ,
                    // 1я скобка - перед символом стоит запятая
                    // 2я скобка - перед текущим символом стоит пробел и перед ним запятая
                    // и мы отрицаем все это дело

                    s.insert(i-1, ","); // вот так оно выглядит xx, а
                    i++;
                }
            }
        }

        if (s.substr(i, 3) == "but") // нашли but
        {
            if ((i == 0 || s[i - 1] == ' ') &&
                (i + 3 == s.length() || s[i + 3] == ' '))
            {
                // стоим на первом символе but, и слева либо начало строки, либо пробел
                // и (после слова but либо конец строки, либо пробел)
                // тут +3, тк but из 3 символов

                if (!((i > 0 && s[i - 1] == ',') ||
                      (i > 1 && s[i - 2] == ',' && s[i - 1] == ' ')))
                {
                    // если перед словом but нет запятой — вставляем ее
                    // 1-я скобка: перед but сразу стоит запятая
                    // 2-я скобка = если там xx, but
                    // и мы отрицаем все это дело

                    s.insert(i-1, ","); // чтобы выглядело так: xx, but
                    i++;
                }
            }
        }
    }

    cout << s << endl;

    return 0;
}
```
